// SPDX-License-Identifier: MIT pragma solidity ^0.8.20; import "@openzeppelin/contracts/token/ERC20/IERC20.sol"; contract VeloPay { IERC20 public token; uint256 public constant ADVANCE_BPS = 9000; // %90 aninda odenir uint256 public constant LATE_FEE_BPS_PER_DAY = 10; // gecikilen her gun icin %0.1 ceza struct Invoice { address freelancer; address debtor; uint256 faceValue; uint256 dueDate; uint256 advanced; bool repaid; bool defaulted; } uint256 public nextInvoiceId; mapping(uint256 => Invoice) public invoices; event LiquidityAdded(address indexed provider, uint256 amount); event InvoiceCreated(uint256 indexed id, address indexed freelancer, address indexed debtor, uint256 faceValue, uint256 dueDate, uint256 advanced); event InvoiceRepaid(uint256 indexed id, uint256 amountPaid); event InvoiceDefaulted(uint256 indexed id); constructor(address _token) { token = IERC20(_token); } // Yatirimcilar havuza likidite ekler function addLiquidity(uint256 amount) external { token.transferFrom(msg.sender, address(this), amount); emit LiquidityAdded(msg.sender, amount); } // Freelancer fatura olusturur, %90'i aninda odenir function createInvoice(address debtor, uint256 faceValue, uint256 dueDate) external returns (uint256) { require(faceValue > 0, "Miktar sifir olamaz"); require(dueDate > block.timestamp, "Vade gelecekte olmali"); uint256 advanceAmount = (faceValue * ADVANCE_BPS) / 10000; require(token.balanceOf(address(this)) >= advanceAmount, "Havuzda yeterli likidite yok"); uint256 id = nextInvoiceId++; invoices[id] = Invoice({ freelancer: msg.sender, debtor: debtor, faceValue: faceValue, dueDate: dueDate, advanced: advanceAmount, repaid: false, defaulted: false }); token.transfer(msg.sender, advanceAmount); emit InvoiceCreated(id, msg.sender, debtor, faceValue, dueDate, advanceAmount); return id; } // Su an odenmesi gereken tutar (vade gecmisse ceza dahil) function getAmountDue(uint256 id) public view returns (uint256) { Invoice memory inv = invoices[id]; if (block.timestamp <= inv.dueDate) return inv.faceValue; uint256 daysLate = (block.timestamp - inv.dueDate) / 1 days; uint256 penalty = (inv.faceValue * LATE_FEE_BPS_PER_DAY * daysLate) / 10000; return inv.faceValue + penalty; } // Borclu faturayi kapatir (vade gecmisse ceza dahil oder) function repayInvoice(uint256 id) external { Invoice storage inv = invoices[id]; require(!inv.repaid && !inv.defaulted, "Zaten kapatilmis"); uint256 amountDue = getAmountDue(id); token.transferFrom(msg.sender, address(this), amountDue); inv.repaid = true; emit InvoiceRepaid(id, amountDue); } // Vade + 7 gun sonra hala odenmemisse, herkes temerrut olarak isaretleyebilir function markDefaulted(uint256 id) external { Invoice storage inv = invoices[id]; require(!inv.repaid && !inv.defaulted, "Zaten kapatilmis"); require(block.timestamp > inv.dueDate + 7 days, "Henuz temerrut suresi dolmadi"); inv.defaulted = true; emit InvoiceDefaulted(id); } function getInvoice(uint256 id) external view returns (address, address, uint256, uint256, uint256, bool, bool) { Invoice memory inv = invoices[id]; return (inv.freelancer, inv.debtor, inv.faceValue, inv.dueDate, inv.advanced, inv.repaid, inv.defaulted); } }
